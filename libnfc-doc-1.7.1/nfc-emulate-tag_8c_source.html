00084 static bool
00085 target_io(nfc_target *pnt, const uint8_t *pbtInput, const size_t szInput, uint8_t *pbtOutput, size_t *pszOutput)
00086 {
00087   bool loop = true;
00088   *pszOutput = 0;
00089 
00090   // Show transmitted command
00091   if (!quiet_output) {
00092     printf("    In: ");
00093     print_hex(pbtInput, szInput);
00094   }
00095   if (szInput) {
00096     switch (pbtInput[0]) {
00097       case 0x30: // Mifare read
00098         // block address is in pbtInput[1]
00099         *pszOutput = 15;
00100         strcpy((char *)pbtOutput, "You read block ");
00101         pbtOutput[15] = pbtInput[1];
00102         break;
00103       case 0x50: // HLTA (ISO14443-3)
00104         if (!quiet_output) {
00105           printf("Initiator HLTA me. Bye!\n");
00106         }
00107         loop = false;
00108         break;
00109       case 0x60: // Mifare authA
00110       case 0x61: // Mifare authB
00111         // Let's give back a very random nonce...
00112         *pszOutput = 2;
00113         pbtOutput[0] = 0x12;
00114         pbtOutput[1] = 0x34;
00115         // Next commands will be without CRC
00116         init_mfc_auth = true;
00117         break;
00118       case 0xe0: // RATS (ISO14443-4)
00119         // Send ATS
00120         *pszOutput = pnt->nti.nai.szAtsLen + 1;
00121         pbtOutput[0] = pnt->nti.nai.szAtsLen + 1; // ISO14443-4 says that ATS contains ATS_Length as first byte
00122         if (pnt->nti.nai.szAtsLen) {
00123           memcpy(pbtOutput + 1, pnt->nti.nai.abtAts, pnt->nti.nai.szAtsLen);
00124         }
00125         break;
00126       case 0xc2: // S-block DESELECT
00127         if (!quiet_output) {
00128           printf("Initiator DESELECT me. Bye!\n");
00129         }
00130         loop = false;
00131         break;
00132       default: // Unknown
00133         if (!quiet_output) {
00134           printf("Unknown frame, emulated target abort.\n");
00135         }
00136         loop = false;
00137     }
00138   }
00139   // Show transmitted command
00140   if ((!quiet_output) && *pszOutput) {
00141     printf("    Out: ");
00142     print_hex(pbtOutput, *pszOutput);
00143   }
00144   return loop;
00145 }

00147 static bool
00148 nfc_target_emulate_tag(nfc_device *dev, nfc_target *pnt)
00149 {
00150   size_t szTx;
00151   uint8_t abtTx[MAX_FRAME_LEN];
00152   bool loop = true;
00153 
00154   if ((szRx = nfc_target_init(dev, pnt, abtRx, sizeof(abtRx), 0)) < 0) {
00155     nfc_perror(dev, "nfc_target_init");
00156     return false;
00157   }
00158 
00159   while (loop) {
00160     loop = target_io(pnt, abtRx, (size_t) szRx, abtTx, &szTx);
00161     if (szTx) {
00162       if (nfc_target_send_bytes(dev, abtTx, szTx, 0) < 0) {
00163         nfc_perror(dev, "nfc_target_send_bytes");
00164         return false;
00165       }
00166     }
00167     if (loop) {
00168       if (init_mfc_auth) {
00169         if (nfc_device_set_property_bool(dev, NP_HANDLE_CRC, false) < 0) {
00170           nfc_perror(pnd, "nfc_target_emulate_tag");
00171           nfc_close(pnd);
00172           nfc_exit(context);
00173           exit(EXIT_FAILURE);
00174         }
00175         init_mfc_auth = false;
00176       }
00177       if ((szRx = nfc_target_receive_bytes(dev, abtRx, sizeof(abtRx), 0)) < 0) {
00178         nfc_perror(dev, "nfc_target_receive_bytes");
00179         return false;
00180       }
00181     }
00182   }
00183   return true;
00184 }
