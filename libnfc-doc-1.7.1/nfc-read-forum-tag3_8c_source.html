00105 nfc_forum_tag_type3_check(nfc_device *dev, const nfc_target *nt, const uint16_t block, const uint8_t block_count, uint8_t *data, size_t *data_len)
00106 {
00107   uint8_t payload[1024] = {
00108     1, // Services
00109     0x0B, 0x00, // NFC Forum Tag Type 3's Service code
00110     block_count,
00111     0x80, block, // block 0
00112   };
00113 
00114   size_t payload_len = 1 + 2 + 1;
00115   for (uint8_t b = 0; b < block_count; b++) {
00116     if (block < 0x100) {
00117       payload[payload_len++] = 0x80;
00118       payload[payload_len++] = block + b;
00119     } else {
00120       payload[payload_len++] = 0x00;
00121       payload[payload_len++] = (block + b) >> 8;
00122       payload[payload_len++] = (block + b) & 0xff;
00123     }
00124   }
00125 
00126   uint8_t frame[1024];
00127   size_t frame_len = sizeof(frame);
00128   build_felica_frame(nt->nti.nfi, CHECK, payload, payload_len, frame, &frame_len);
00129 
00130   uint8_t rx[1024];
00131   int res;
00132   if ((res = nfc_initiator_transceive_bytes(dev, frame, frame_len, rx, sizeof(rx), 0)) < 0) {
00133     return res;
00134   }
00135 
00136   const int res_overhead = 1 + 1 + 8 + 2;  // 1+1+8+2: LEN + CMD + NFCID2 + STATUS
00137   if (res < res_overhead) {
00138     // Not enough data
00139     return -1;
00140   }
00141   uint8_t felica_res_len = rx[0];
00142   if (res != felica_res_len) {
00143     // Error while receiving felica frame
00144     return -1;
00145   }
00146   if ((CHECK + 1) != rx[1]) {
00147     // Command return does not match
00148     return -1;
00149   }
00150   if (0 != memcmp(&rx[2], nt->nti.nfi.abtId, 8)) {
00151     // NFCID2 does not match
00152     return -1;
00153   }
00154   const uint8_t status_flag1 = rx[10];
00155   const uint8_t status_flag2 = rx[11];
00156   if ((status_flag1) || (status_flag2)) {
00157     // Felica card's error
00158     fprintf(stderr, "Status bytes: %02x, %02x\n", status_flag1, status_flag2);
00159     return -1;
00160   }
00161   // const uint8_t res_block_count = res[12];
00162   *data_len = res - res_overhead + 1; // +1 => block count is stored on 1 byte
00163   memcpy(data, &rx[res_overhead + 1], *data_len);
00164   return *data_len;
00165 }
