00140 
00141   printf("Writing %d pages |", uiBlocks + 1);
00142 
00143   // Skip block 0 - as far as I know there are no Jewel tags with block 0 writeable
00144   printf("s");
00145   uiSkippedBlocks++;
00146 
00147   for (block = uiSkippedBlocks; block <= uiBlocks; block++) {
00148     // Skip block 0x0D - it is reserved for internal use and can't be written
00149     if (block == 0x0D) {
00150       printf("s");
00151       uiSkippedBlocks++;
00152       continue;
00153     }
00154     // Skip block 0X0E if lock-bits and OTP shouldn't be written
00155     if ((block == 0x0E) && (!write_lock) && (!write_otp)) {
00156       printf("s");

00158       continue;
00159     }
00160     // Write block 0x0E partially if lock-bits or OTP shouldn't be written
00161     if ((block == 0x0E) && (!write_lock || !write_otp)) {
00162       printf("p");

00164     }
00165 
00166     for (byte = 0; byte < uiBytesPerBlock; byte++) {
00167       if ((block == 0x0E) && (byte == 0 || byte == 1) && (!write_lock)) {
00168         continue;
00169       }
00170       if ((block == 0x0E) && (byte > 1) && (!write_otp)) {
00171         continue;
00172       }
00173 
00174       // Show if the readout went well
00175       if (bFailure) {
00176         // When a failure occured we need to redo the anti-collision
00177         if (nfc_initiator_select_passive_target(pnd, nmJewel, NULL, 0, &nt) <= 0) {
00178           ERR("tag was removed");
00179           return false;
00180         }
00181         bFailure = false;
00182       }
00183 
00184       req.writee.btCmd = TC_WRITEE;
00185       req.writee.btAdd = (block << 3) + byte;
00186       req.writee.btDat = ttDump.ttd.abtData[(block << 3) + byte];
00187       if (!nfc_initiator_jewel_cmd(pnd, req, &res)) {
00188         bFailure = true;
00189       }
00190     }
00191     print_success_or_failure(bFailure, &uiWrittenBlocks);
00192     fflush(stdout);
00193   }
00194   printf("|\n");
